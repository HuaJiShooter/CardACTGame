CREATE TABLE Cards (card_ID      INTEGER PRIMARY KEY AUTOINCREMENT,card_name    VARCHAR(64)  NOT NULL,description  TEXT,cost         INTEGER      DEFAULT 0,card_face    TEXT,cold_time    REAL         DEFAULT 0,cast_time    REAL         DEFAULT -1,modifier_list JSON,object_list   JSON);

CREATE TABLE Modifiers (modifier_ID     INTEGER PRIMARY KEY AUTOINCREMENT,modifier_name   VARCHAR(64) NOT NULL,description     TEXT,condition_list  JSON,effect_list     JSON);
